Colors

Colors can be used to define a color of a role. Corde provide enums with some colors that can be used to help development:

enum ColorsHex {
DEFAULT = 0x000000,
WHITE = 0xffffff,
AQUA = 0x1abc9c,
GREEN = 0x2ecc71,
BLUE = 0x3498db,
YELLOW = 0xffff00,
PURPLE = 0x9b59b6,
LUMINOUS_VIVID_PINK = 0xe91e63,
GOLD = 0xf1c40f,
ORANGE = 0xe67e22,
RED = 0xe74c3c,
GREY = 0x95a5a6,
NAVY = 0x34495e,
DARK_AQUA = 0x11806a,
DARK_GREEN = 0x1f8b4c,
DARK_BLUE = 0x206694,
DARK_PURPLE = 0x71368a,
DARK_VIVID_PINK = 0xad1457,
DARK_GOLD = 0xc27c0e,
DARK_ORANGE = 0xa84300,
DARK_RED = 0x992d22,
DARK_GREY = 0x979c9f,
DARKER_GREY = 0x7f8c8d,
LIGHT_GREY = 0xbcc0c0,
DARK_NAVY = 0x2c3e50,
BLURPLE = 0x7289da,
GREYPLE = 0x99aab5,
DARK_BUT_NOT_BLACK = 0x2c2f33,
NOT_QUITE_BLACK = 0x23272a,
}
enum Colors {
DEFAULT = "#000000",
WHITE = "#FFFFFF",
AQUA = "#1ABC9C",
GREEN = "#2ECC71",
BLUE = "#3498DB",
PURPLE = "#9B59B6",
GOLD = "#E91E63",
ORANGE = "#E67E22",
RED = "#E74C3C",
GREY = "#95A5A6",
DARKER_GREY = "#7F8C8D",
NAVY = "#34495E",
DARK_AQUA = "#11806A",

DARK_BLUE = "#206694",
DARK_PURPLE = "#71368A",
DARK_GOLD = "#C27C0E",
DARK_ORANGE = "#A84300",
DARK_RED = "#992D22",
DARK_GREY = "#7F8C8D",

These values can be accessed with

import { Colors, ColorsHex } from "corde";
